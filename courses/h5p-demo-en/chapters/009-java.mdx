```java question:java-demo starter
public class Main {
    public static void main(String[] args) {
        int n = 27;
        int schritte = 0;

        System.out.print("Collatz sequence for " + n + ": ");
        System.out.print(n);

        while (n != 1) {
            if (n % 2 == 0) {
                n = n / 2;
            } else {
                n = 3 * n + 1;
            }
            System.out.print(" → " + n);
            schritte++;
        }

        System.out.println();
        System.out.println("Steps: " + schritte);
    }
}
```
