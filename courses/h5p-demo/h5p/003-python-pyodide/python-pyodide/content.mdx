```python grading:targetCode
import numpy as np

messwerte = np.array([12.5, 14.2, 11.8, 15.0, 13.7, 12.1, 14.8, 11.5, 13.9, 14.4])

mittelwert = np.mean(messwerte)
median = np.median(messwerte)
std = np.std(messwerte)

print(f"Mittelwert: {mittelwert:.2f}")
print(f"Median:     {median:.2f}")
print(f"Std-Abw.:   {std:.2f}")
```
