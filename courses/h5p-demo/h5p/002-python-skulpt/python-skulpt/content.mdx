def ist_primzahl(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

primzahlen = [n for n in range(2, 51) if ist_primzahl(n)]
print("Primzahlen bis 50:")
print(primzahlen)
print(f"Anzahl: {len(primzahlen)}")
